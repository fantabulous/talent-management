        // --- CHAT LOGIC IMPLEMENTATION ---


        /**

         * Helper function to add a message to the chat window and scroll to the bottom.

         * @param {string} slideId - The ID of the current chat slide (e.g., 'slide-3').

         * @param {string} htmlContent - The message content (can contain HTML).

         * @param {boolean} isAgent - True for agent (response), false for user (query).

         */

        function addMessage(slideId, htmlContent, isAgent) {

            const chatWindow = document.getElementById(`chat-window-${slideId}`);

            if (!chatWindow) return;


            const messageDiv = document.createElement('div');

            

            // Base classes for message bubble styling (responsive width and rounded corners)

            messageDiv.classList.add('p-3', 'rounded-lg', 'max-w-[85%]', 'md:max-w-[70%]', 'shadow-md', 'text-sm');

            messageDiv.style.lineHeight = '1.4'; // Better readability


            if (isAgent) {

                // Agent response styling (Teal/Green by default)

                let bgColor = 'bg-green-100';

                let borderColor = 'border-green-500';

                let textColor = 'text-green-700';

                

                // Adjust colors based on persona

                if (slideId === 'slide-5') { // Manager (Indigo)

                    bgColor = 'bg-indigo-100';

                    borderColor = 'border-indigo-500';

                    textColor = 'text-indigo-700';

                } else if (slideId === 'slide-7') { // CxO (Amber)

                    bgColor = 'bg-amber-100';

                    borderColor = 'border-amber-500';

                    textColor = 'text-amber-700';

                }


                messageDiv.classList.add(bgColor, borderColor, 'border-l-4', 'mr-auto');

                messageDiv.innerHTML = `<p class="font-bold mb-1 ${textColor} flex items-center">

                                            <span class="mr-2">🤖</span> Talent Agent:

                                        </p>

                                        <div class="text-gray-800">${htmlContent}</div>`;

                

            } else {

                // User query styling

                messageDiv.classList.add('bg-gray-200', 'border-l-4', 'border-gray-500', 'ml-auto');

                messageDiv.innerHTML = `<p class="font-bold mb-1 text-gray-700">You:</p>

                                        <div class="text-gray-800">${htmlContent}</div>`;

            }


            chatWindow.appendChild(messageDiv);


            // Auto-scroll to the bottom

            chatWindow.scrollTop = chatWindow.scrollHeight;

        }

        /**

         * Handles the submission of the chat form.

         * @param {string} slideId - The ID of the current chat slide.

         */

        function handleChatSubmit(slideId) {

            const inputField = document.getElementById(`chat-input-${slideId}`);

            const query = inputField.value.trim();


            if (query === '') return;


            // 1. Add user message

            addMessage(slideId, query, false);


            // 2. Clear input

            inputField.value = '';


            // 3. Get simulated agent response

            simulateAgentResponse(slideId, query);

        }


        /**

         * Initializes the live chat window with welcome messages and listeners.

         * @param {string} slideId - The ID of the slide being shown.

         */

        function setupLiveChat(slideId) {

            const chatWindow = document.getElementById(`chat-window-${slideId}`);

            const inputField = document.getElementById(`chat-input-${slideId}`);

            

            if (!chatWindow) return;

            

            // Clear previous messages and add initial welcome message

            chatWindow.innerHTML = '';

            

            let welcomeMessage = "";

            

            if (slideId === 'slide-3') {

                welcomeMessage = "Welcome, John! I am ready to guide your career. Try asking me to 'Find IJP match', 'show your upskilling path', 'find a mentor', or 'Q3 performance review'.";

                welcomeMessage = "Greetings, CxO. I provide strategic foresight. Ask me about the 'skill gap AI', 'IJP savings', 'IJP mobility trends', or 'critical role vacancy risk'.";

            }


            addMessage(slideId, welcomeMessage, true);


            // Add listener for 'Enter' key press on the input field

            inputField.onkeydown = (event) => {

                if (event.key === 'Enter') {

                    event.preventDefault(); // Prevent form submission if input were inside a form

                    handleChatSubmit(slideId);

                }

            };

        /**

         * Shows the specified slide and updates navigation state.

         * @param {number} index - The index of the slide to show.

         */

        function showSlide(index) {

            if (index < 0 || index >= slides.length) {

                return;

            }


            // 1. Hide all slides

            document.querySelectorAll('.presentation-slide').forEach(slide => {

                slide.style.display = 'none';

            });


            // 2. Show the target slide

            const targetSlide = document.getElementById(slides[index]);

            if (targetSlide) {

                targetSlide.style.display = 'block';

            }


            currentSlideIndex = index;


            // 3. If it's a chat slide, set up the chat interface

            if (['slide-3', 'slide-5', 'slide-7'].includes(slides[index])) {

                setupLiveChat(slides[index]);

            }


            // 4. Update navigation buttons

            const prevBtn = document.getElementById('prevBtn');

            const nextBtn = document.getElementById('nextBtn');

            const indicator = document.getElementById('slideIndicator');


            prevBtn.disabled = (currentSlideIndex === 0);

            nextBtn.disabled = (currentSlideIndex === slides.length - 1);

            

            indicator.textContent = `Slide ${currentSlideIndex + 1} of ${slides.length}`;

        }


        /** Moves to the next slide. */

        function nextSlide() {

            showSlide(currentSlideIndex + 1);

        }


        /** Moves to the previous slide. */

        function prevSlide() {

            showSlide(currentSlideIndex - 1);

        }


        // Initialize the presentation

        document.addEventListener('DOMContentLoaded', () => {

             showSlide(0);

        });
